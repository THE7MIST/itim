🔹 STEP 5: Add Pipeline Script

pipeline {
    agent any

    stages {

        stage('Clone') {
            steps {
                git 'https://github.com/<username>/<repo>.git'
            }
        }

        stage('Build') {
            steps {
                sh 'javac HelloWorld.java'
            }
        }

        stage('Run') {
            steps {
                sh 'java HelloWorld'
            }
        }
    }
}

🔹 STEP 12: Update Pipeline

environment {
    IMAGE = "<your-dockerhub-username>/java-app:v1"
}

stage('Tag Image') {
    steps {
        sh 'docker tag java-app $IMAGE'
    }
}

stage('Push Image') {
    steps {
        withCredentials([usernamePassword(credentialsId: 'dockerhub', usernameVariable: 'USER', passwordVariable: 'PASS')]) {
            sh '''
            echo $PASS | docker login -u $USER --password-stdin
            docker push $IMAGE
            '''
        }
    }
}

🟣 PART 4 — Deploy to Kubernetes using Jenkins


---

🔹 STEP 14: Create deployment.yaml

apiVersion: apps/v1
kind: Deployment
metadata:
  name: java-app
spec:
  replicas: 2
  selector:
    matchLabels:
      app: java-app
  template:
    metadata:
      labels:
        app: java-app
    spec:
      containers:
      - name: java-app
        image: <your-dockerhub-username>/java-app:v1


---

🔹 STEP 15: Create service.yaml

apiVersion: v1
kind: Service
metadata:
  name: java-service
spec:
  type: NodePort
  selector:
    app: java-app
  ports:
    - port: 80
      targetPort: 8080
      nodePort: 30007
